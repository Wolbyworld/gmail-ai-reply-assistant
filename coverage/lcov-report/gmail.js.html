/**
 * Returns a static NodeList of Gmail compose window root elements.
 * Searches for elements with aria-label="Message Body" that contain a child
 * with the class 'editable'. Note: Gmail's DOM can change; this selector
 * might need future adjustments.
 */
export function getComposeWindows() {
  // Gmail's compose window structure seems to involve a container for the message body
  // with a specific aria-label, and inside it, the editable area.
  // This selector attempts to find the 'Message Body' container element.
  // It might need refinement based on actual Gmail DOM inspection.
  const composeWindows = document.querySelectorAll('[aria-label="Message Body"]');
 
  // Further filter or return based on the presence of '.editable' if necessary,
  // although the aria-label might be sufficient if it's unique to the editable body container.
  // For simplicity and based on the prompt's core requirement, we'll rely on the aria-label for now.
  // A more robust approach might involve finding the editable and traversing up.
  return composeWindows;
}
